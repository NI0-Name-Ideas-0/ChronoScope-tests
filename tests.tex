\textbf{Testdaten:} Montag: 09:00--17:00, Dienstag: 10:00--16:00, Mittwoch: 09:00--17:00, Donnerstag: 10:00--16:00, Freitag: 09:00--15:00.

\begin{longtable}{cp{5.5cm}p{5.5cm}}

    \textbf{Nr.} & \textbf{Aktion} & \textbf{Erwartetes Ergebnis} \\
    \midrule

    2 & Auf den Reiter \glqq Arbeitszeiten\grqq \ klicken & Der Reiter wird angezeigt \\
    3 & Arbeitszeiten per Drag-und-Drop hinzufügen, verschieben, löschen & Arbeitszeiten werden angezeigt \\
