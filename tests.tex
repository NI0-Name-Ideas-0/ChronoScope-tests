\subsection{OR-49 -- Kalender, Listen und Split View}
% ------------------------------------------------------------

\subsubsection*{ST-OR49-01: Kalenderansicht laden}

\textbf{Testziel:} Kalenderansicht wird korrekt geladen und Aufgaben werden angezeigt.

\textbf{Akzeptanzkriterien:} Ladezeit $<$ 5\,s (AK010); Oberfläche konsistent (NFA030--NFA031)

\begin{longtable}{cp{5.5cm}p{5.5cm}}
    \toprule
    \textbf{Nr.} & \textbf{Aktion} & \textbf{Erwartetes Ergebnis} \\
    \midrule
    1 & Benutzer navigiert zur Kalenderansicht & Kalender wird angezeigt \\
    2 & Benutzer prüft, ob vorhandene Aufgaben sichtbar sind & Aufgaben erscheinen im Kalender \\
    3 & Benutzer wechselt zwischen Tages-/Wochen-/Monatsansicht & Ansicht wechselt korrekt \\
    \bottomrule
\end{longtable}

\bigskip

\subsubsection*{ST-OR49-02: Listenansicht anzeigen}

\textbf{Testziel:} Listenansicht zeigt alle Tasks (statisch und dynamisch) an.

\begin{longtable}{cp{5.5cm}p{5.5cm}}
    \toprule
    \textbf{Nr.} & \textbf{Aktion} & \textbf{Erwartetes Ergebnis} \\
    \midrule
    1 & Benutzer wechselt zur Listenansicht & Liste aller Aufgaben wird angezeigt \\
    2 & Benutzer prüft, ob statische und dynamische Tasks vorhanden sind & Beide Typen sind sichtbar \\
    \bottomrule
\end{longtable}

\bigskip

\subsubsection*{ST-OR49-03: Split View verwenden}

\textbf{Testziel:} Split View zeigt Kalender und Liste gleichzeitig an.

\begin{longtable}{cp{5.5cm}p{5.5cm}}
    \toprule
    \textbf{Nr.} & \textbf{Aktion} & \textbf{Erwartetes Ergebnis} \\
    \midrule
    1 & Benutzer aktiviert Split View & Kalender und Liste werden nebeneinander angezeigt \\
    2 & Benutzer erstellt eine Aufgabe & Aufgabe erscheint in beiden Ansichten \\
    \bottomrule
\end{longtable}
