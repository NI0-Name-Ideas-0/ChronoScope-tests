\subsection{OR-49 -- Kalender, Listen und Split View}
% ------------------------------------------------------------

\subsubsection*{ST-OR49-01: Kalenderansicht laden}

\textbf{Testziel:} Kalenderansicht wird korrekt geladen und Aufgaben werden angezeigt.

\textbf{Akzeptanzkriterien:} Ladezeit $<$ 5\,s (AK010); Oberfl\"ache konsistent (NFA030--NFA031)

\begin{longtable}{cp{5.5cm}p{5.5cm}}
\toprule
\textbf{Nr.} & \textbf{Aktion} & \textbf{Erwartetes Ergebnis} \\
\midrule
1 & Benutzer navigiert zur Kalenderansicht & Kalender wird angezeigt \\
2 & Benutzer pr\"uft, ob vorhandene Aufgaben sichtbar sind & Aufgaben erscheinen im Kalender \\
3 & Benutzer wechselt zwischen Tages-/Wochen-/Monatsansicht & Ansicht wechselt korrekt \\
\bottomrule
\end{longtable}

\textbf{Ergebnis:} \rule{10cm}{0.4pt}

\bottomrule
\end{longtable}

\textbf{Ergebnis:} \rule{10cm}{0.4pt}

\subsection{OR-50 -- Aufgaben/Termine/Blocker erfassen}
% ------------------------------------------------------------

\subsubsection*{ST-OR50-01: Dynamische Aufgabe erfolgreich erfassen}

\textbf{Testziel:} Benutzer kann eine dynamische Aufgabe erstellen und diese wird gespeichert.

\textbf{Akzeptanzkriterien:} Fehlereingaben werden erkannt; Kommunikationsfehler behandelt

\textbf{Testdaten:}
\begin{tabular}{@{}ll@{}}
Name & \textit{Bericht schreiben} \\
Beschreibung & \textit{Schreibe einen Bericht dar\"uber} \\
Deadline & 06.07.2026 \\
Dauer & 120\,min \\
\end{tabular}

\begin{longtable}{cp{5.5cm}p{5.5cm}}
\toprule
\textbf{Nr.} & \textbf{Aktion} & \textbf{Erwartetes Ergebnis} \\
\midrule
1 & Benutzer klickt auf \glqq Aufgabe erstellen\grqq{} & Erstellungsdialog \"offnet sich \\
2 & Benutzer gibt Testdaten ein & Eingaben werden \"ubernommen \\
3 & Benutzer w\"ahlt Typ \glqq Dynamisch\grqq{} & Zus\"atzliche Felder (Dauer, Min-/Max-Scope) erscheinen \\
4 & Benutzer speichert & Aufgabe erscheint in der \"Ubersicht \\
\bottomrule
\end{longtable}

\textbf{Ergebnis:} \rule{10cm}{0.4pt}
